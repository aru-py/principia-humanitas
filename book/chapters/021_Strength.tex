\chapter{Strength}

\begin{equation}
\text{Strength} = (P \times D \times C) + (E \times I) - F
\end{equation}

\textbf{Where:}

\begin{itemize}
    \item $P$: Physical Capability
    \item $D$: Determination
    \item $C$: Compassion
    \item $E$: Experience
    \item $I$: Insight
    \item $F$: Fear
\end{itemize}
