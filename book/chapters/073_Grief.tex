\chapter{Grief}

\begin{equation}
\text{Grief} = \frac{L \times (M + D)}{R} - H
\end{equation}

\textbf{Where:}

\begin{itemize}
    \item $L$: Love for the departed or lost
    \item $M$: Magnitude of the loss
    \item $D$: Duration since the loss occurred
    \item $R$: Resilience of the individual
    \item $H$: Hope for the future
\end{itemize}
